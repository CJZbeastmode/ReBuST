\section{Problem Formulation}

\subsection{Setting}
We are given a Whole Slide Image (WSI) organized as a pyramid with $L$ levels, where level $\ell = 0$ is the finest (highest resolution) and level $\ell = L-1$ is the coarsest.

Inspiration: At each iteration, HUMBE evaluates all currently selected patches, ranks all feasible zoom operations by their marginal utility per added patch, and greedily applies the most efficient refinements in a batched manner until the global budget is reached.

\subsection{Hierarchical Structure}
Patches form a tree structure where:
\begin{itemize}[nosep]
    \item Each patch at level $\ell$ can be \emph{zoomed} into $k$ child patches at level $\ell - 1$
    \item Typically $k = 4$ or $k = 16$ (2×2 or 4x4 subdivision)
    \item A patch can be either \emph{KEPT} (selected as a leaf) or \emph{ZOOMED} (replaced by children)
\end{itemize}

\subsection{Scoring Function}
For each patch $p$, we define:
\begin{itemize}[nosep]
    \item $s_{\text{kept}}(p)$: utility if we keep $p$ as a leaf
    \item $s_{\text{zoom}}(p, C)$: aggregated utility of children $C = \{c_1, \ldots, c_k\}$
\end{itemize}

\begin{align}
s_{\text{kept}}(p) &= \text{TextAlign}(p) = \frac{1}{|K|} \sum_{k \in K} \cos(e_{\text{img}}(p), e_{\text{text}}(k)) \\
s_{\text{zoomed}}(p, C) &= \frac{1}{|C|} \sum_{c \in C} \text{TextAlign}(c)
\end{align}
$K$ = set of pathology-related keywords, $C$ = child patches.

\subsection{Optimization Problem}
Let $S \subseteq \mathcal{P}$ be the set of selected (leaf) patches. We want to solve:
\begin{align}
\max_{S \subseteq \mathcal{P}} \quad & \sum_{p \in S} s_{\text{kept}}(p) \\
\text{subject to} \quad & |S| \leq B \\
& \text{hierarchical feasibility}
\end{align}

where $B$ is the budget (maximum number of patches) and hierarchical feasibility means:
\begin{itemize}[nosep]
    \item If parent $p$ is zoomed, exactly its children are eligible for selection
    \item Cannot keep both a parent and its descendants
\end{itemize}

\subsection{Connection to Knapsack}
This is a variant of the 0/1 Knapsack problem with hierarchical constraints:
\begin{itemize}[nosep]
    \item \textbf{Items}: Refinement operations
    \item \textbf{Value}: Utility gain $g = s_{\text{zoomed}}(p, C) - s_{\text{kept}}(p)$
    \item \textbf{Weight}: Net cost $w = |C| - 1$ (increase in patch count)
    \item \textbf{Capacity}: Budget $B$
    \item \textbf{Constraint}: Hierarchical dependencies (can only apply refinement if parent exists)
\end{itemize}

\section{HUMBE Algorithm}

\subsection{Core Idea}
HUMBE uses a \emph{greedy refinement strategy}:
\begin{enumerate}[nosep]
    \item Start with all root-level patches
    \item Iteratively refine patches with highest efficiency
    \item Stop when budget is used up or no beneficial refinements remain
\end{enumerate}

\subsection{Efficiency Metric}
For a refinement operation replacing parent $p$ with children $C$:
\begin{equation}
\text{Efficiency}(p \to C) = \frac{s_{\text{zoomed}}(p, C) - s_{\text{kept}}(p)}{|C| - 1} = \frac{\text{gain}}{\text{cost}}
\end{equation}

This measures the \emph{utility gain per unit budget consumed}.

\subsection{Pseudocode}
This section sketches the pseudo code of the HUMBE algorithm: (in the next page)

\begin{algorithm}
\caption{HUMBE (Hierarchical Utility-Maximizing Budget Enforcer)}
\begin{algorithmic}[1]
\State \textbf{Input:} WSI pyramid, score function $s(\cdot)$, budget ratio $\rho$, batch size $K$
\State \textbf{Output:} Selected patches $S$, zoomed patches $Z$

\State $S \gets \{$all root-level patches$\}$
\State $Z \gets \emptyset$
\State $B \gets \lfloor \rho \cdot |\text{all patches}| \rfloor$

\While{$|S| < B$}
    \State $\mathcal{C} \gets \emptyset$ \Comment{Candidate refinements}
    
    \For{each patch $p \in S$ with level $\ell > 0$}
        \State $C \gets \text{children}(p)$
        \If{$C \neq \emptyset$}
            \State $g \gets s_{\text{zoomed}}(p, C) - s_{\text{kept}}(p)$
            \State $w \gets |C| - 1$
            \If{$g > 0$ \textbf{and} $|S| + w \leq B$}
                \State $\text{eff} \gets g / w$
                \State Add $(p \to C, g, w, \text{eff})$ to $\mathcal{C}$
            \EndIf
        \EndIf
    \EndFor
    
    \If{$\mathcal{C} = \emptyset$}
        \State \textbf{break} \Comment{No beneficial refinements}
    \EndIf
    
    \State Sort $\mathcal{C}$ by efficiency \Comment{Sort by descending}
    
    \State $\text{applied} \gets 0$ \Comment{No beneficial refinements}
    
    \For{each candidate $(p \to C, g, w, \text{eff})$ in $\mathcal{C}$}
        \If{$\text{applied} \geq K$ \textbf{or} $p \notin S$ \textbf{or} $|S| + w > B$}
            \State \textbf{continue}
        \EndIf
        
        \State $S \gets S \setminus \{p\}$ \Comment{Remove parent}
        \State $Z \gets Z \cup \{p\}$ \Comment{Mark as zoomed}
        \State $S \gets S \cup C$ \Comment{Add children}
        \State $\text{applied} \gets \text{applied} + 1$
    \EndFor
    
    \If{$\text{applied} = 0$}
        \State \textbf{break} \Comment{Could not apply any refinements}
    \EndIf
\EndWhile

\State \Return $S, Z$ \Comment{Return kept and zoomed (for visualization purpose)}
\end{algorithmic}
\end{algorithm}

\subsection{Key Features}

\begin{itemize}
    \item \textbf{Greedy Selection}: Chooses refinements with highest benefit-per-cost ratio
    \item \textbf{Batch Processing}: Applies top-$K$ refinements per iteration to avoid local optima
    \item \textbf{Non-Conflicting}: Only applies refinements where parent still exists
    \item \textbf{Budget-Aware}: Budget constraint is strictly respected
\end{itemize}

\section{Theoretical Analysis}

\subsection{Time Complexity}
\begin{itemize}[nosep]
    \item Initialization: $O(n_{\text{root}})$ where $n_{\text{root}}$ is the number of root patches
    \item Per iteration:
    \begin{itemize}[nosep]
        \item Evaluate candidates: $O(|S| \cdot k)$ where $k$ is branching factor
        \item Sort candidates: $O(|\mathcal{C}| \log |\mathcal{C}|)$
        \item Apply refinements: $O(K)$
    \end{itemize}
    \item Number of iterations: $O(\log_{k}(B / n_{\text{root}}))$ (pyramid depth)
    \item \textbf{Total}: $O(B \cdot k \cdot \log B)$
\end{itemize}

\subsection{Space Complexity}
$O(B)$ for storing selected patches and candidates.

\subsection{Approximation Guarantee}
\begin{itemize}[nosep]
    \item For standard fractional knapsack: greedy is \textbf{optimal}
    \item For 0/1 knapsack with hierarchy: greedy has worst-case $\frac{1}{2}$-approximation
    \item In practice: performs near-optimally due to spatial correlation in patch scores
\end{itemize}

\section{Example}

\subsection{Setup}
\begin{itemize}[nosep]
    \item WSI with 3 levels: $\ell \in \{2, 1, 0\}$ (coarse to fine)
    \item Root level 2: 4 patches (2×2 grid)
    \item Each patch subdivides into 4 children (2×2)
    \item Budget: $B = 10$ patches
\end{itemize}

\subsection{Initialization}
Start with all root patches:
\begin{align*}
S &= \{p_{2,0,0}, p_{2,0,1}, p_{2,1,0}, p_{2,1,1}\}, \quad |S| = 4 \\
s_{\text{kept}}(p_{2,0,0}) &= 44.0, \quad s_{\text{kept}}(p_{2,0,1}) = 36.0 \\
s_{\text{kept}}(p_{2,1,0}) &= 35.5, \quad s_{\text{kept}}(p_{2,1,1}) = 37.0
\end{align*}

\subsection{Iteration 1: Evaluate Candidates}

For patch $p_{2,0,0}$ (high score):
\begin{align*}
C &= \{p_{1,0,0}, p_{1,0,1}, p_{1,1,0}, p_{1,1,1}\} \\
s_{\text{kept}}(c\in C) &= \{44.3, 44.6, 43.8, 44.1\} \\
s_{\text{zoomed}}(p_{2,0,0}, C) &= \frac{1}{4}(44.3 + 44.6 + 43.8 + 44.1) = 44.2 \\
g &= 44.2 - 44.0 = 0.2 \\
w &= 4 - 1 = 3 \\
\text{eff} &= \frac{0.2}{3} \approx 0.067
\end{align*}

For patch $p_{2,0,1}$ (lower score):
\begin{align*}
C &= \{p_{1,0,2}, p_{1,0,3}, p_{1,1,2}, p_{1,1,3}\} \\
s_{\text{kept}}(c\in C) &= \{41.0, 42.5, 41.8, 40.6\} \\
s_{\text{zoomed}}(p_{2,0,1}, C) &= \frac{1}{4}(41.0 + 42.5 + 41.8 + 40.6) = 41.5 \\
g &= 41.5 - 36.0 = 5.5 \\
w &= 3 \\
\text{eff} &= \frac{5.5}{3} \approx 1.83 \quad \textbf{(highest)}
\end{align*}

\subsection{Apply Best Refinement}
Sort by efficiency: $p_{2,0,1}$ is best.

Apply refinement:
\begin{align*}
S &\gets S \setminus \{p_{2,0,1}\} \cup \{p_{1,0,2}, p_{1,0,3}, p_{1,1,2}, p_{1,1,3}\} \\
Z &\gets \{p_{2,0,1}\} \\
|S| &= 4 - 1 + 4 = 7
\end{align*}

\textbf{Insight}: Although $p_{2,0,1}$ has a relatively low coarse-level score, zooming reveals substantially higher-scoring children, yielding the highest utility gain per unit budget.

\subsection{Iteration 2}
Continue refinements until $|S| = 10$ or no beneficial refinements remain.

\subsection{Final Result}
\begin{itemize}[nosep]
    \item Selected: 10 patches (mix of levels 1 and 2)
    \item Zoomed: Patches with high gain-to-cost ratio
    \item Total utility: Maximized subject to budget
\end{itemize}

\section{Conclusion}

HUMBE is a principled, efficient algorithm for budgeted hierarchical patch selection:

\begin{itemize}
    \item \textbf{Goal}: To locally estimate the most effective patches under a constraint budget
    \item \textbf{Mathematics}: Based on greedy knapsack approximation
    \item \textbf{Computation}: $O(B \log B)$ time complexity
    \item \textbf{Flexibility}: Works with any scoring function $s(\cdot)$
\end{itemize}

The algorithm automatically discovers high-resolution regions of interest while respecting the budget constraint, making it ideal for large-scale WSI analysis.
